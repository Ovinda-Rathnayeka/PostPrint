            // JavaScript function to toggle the visibility of the text box
            function toggleTextBox(id) {
                var textBox = document.getElementById("customerNum_" + id);
                var imageDiv = document.getElementById("cusdil_" + id);

                // Check if the clicked text box is already open
                if (textBox.style.display === "block") {
                    textBox.style.display = "none";
                    imageDiv.style.display = "none";
                } else {
                    // Close all text boxes first
                    var allTextBoxes = document.querySelectorAll("[id^='customerNum_']");
                    var allImageDivs = document.querySelectorAll("[id^='cusdil_']");

                    for (var i = 0; i < allTextBoxes.length; i++) {
                        allTextBoxes[i].style.display = "none";
                    }
                    for (var i = 0; i < allImageDivs.length; i++) {
                        allImageDivs[i].style.display = "none";
                    }

                    // Open the clicked text box
                    textBox.style.display = "block";
                    imageDiv.style.display = "block";
                }
            }

            var cash = document.getElementById('cashamo' + i).value;

            document.getElementById('balan' + i).value = x.toFixed(2);

        function calcard(i) {
            var tot = parseFloat(document.getElementById('totamo' + i).value);
            var card = parseFloat(document.getElementById('cardamo' + i).value);

            if (tot >= card) {
                var x = tot - card;
                var cash = document.getElementById('cashamo' + i).value = x.toFixed(2);
                calball(i);
            } else {
                document.getElementById('cardamo' + i).value = '0.00';
                document.getElementById('cashamo' + i).value = tot;
                calball(i);
            }
        }

        function selectnum() {

        function checktot() {

            var tot = document.getElementById('netpay').value;
            var cash = document.getElementById('cash').value;
            var card = document.getElementById('card').value;
            var bal = document.getElementById('bal');

            if (cusType == 'registered') {
                var change = Number(tot) - (Number(cash) + Number(card));

                if (change > 0) {
                    bal.value = change.toFixed(2);
                } else {
                    bal.value = '0.00';

                if (change > 0) {
                    bal.value = change.toFixed(2);
                } else {
                    bal.value = '0.00';
                }
            }
        }

        function isNumberKey(evt) {
            var charCode = (evt.which) ? evt.which : event.keyCode
            if (charCode > 31 && (charCode < 46 || charCode > 57))
                return false;

            return true;
        }

        jQuery.fn.flash = function(color, duration) {

            for (var i = 0; i < 1000; i++) {

                var current = this.css('backgroundColor');

                this.animate({
                    backgroundColor: 'rgb(' + color + ')'
                }, duration / 2);
                this.animate({
                    backgroundColor: current
                }, duration / 2);
            }

        }

        function calculateDue() {
            var total = parseFloat(document.getElementById('tot').value);
            var payment = parseFloat(document.getElementById('pay').value);

            // Check if payment is greater than total
            if (payment > total) {
                alert("Payment cannot be greater than total amount.");
                document.getElementById('pay').value = ""; // Clear the payment input box
                return;
            }

            var due = total - payment;
            // Set due to 0 if it's negative
            due = (due < 0 || due === total) ? 0 : due;
            // Update the value of the 'due' input box
            document.getElementById('due').value = due.toFixed(2);
        }

        function calculateDue2(i) {
            var total = parseFloat(document.getElementById('tot' + i).value);
            var advance = parseFloat(document.getElementById('a_pay' + i).value);
            var payment = parseFloat(document.getElementById('pay' + i).value);

            // Check if payment is greater than total
            if (payment > (total - advance)) {
                alert("Payment cannot be greater than total amount.");
                document.getElementById('pay<?php echo $i; ?>').value = ""; // Clear the payment input box
                return;
            }

            //var amount = total - advance;
            var due = total - advance - payment;
            // Set due to 0 if it's negative
            due = (due < 0 || due === total) ? 0 : due;
            // Update the value of the 'due' input box
            document.getElementById('due' + i).value = due.toFixed(2);
        }

        function calculateDue3(i) {
            var total = parseFloat(document.getElementById('tot' + i).value);
            var advance = parseFloat(document.getElementById('a_pay' + i).value);
            var payment1 = parseFloat(document.getElementById('payC' + i).value) || 0;
            var payment2 = parseFloat(document.getElementById('payD' + i).value) || 0;

            // Check if payment is greater than total
            if ((payment1 + payment2) > (total - advance)) {
                alert("Payment cannot be greater than total amount.");
                document.getElementById('payC<?php echo $i; ?>').value = ""; // Clear the payment input box
                document.getElementById('payD<?php echo $i; ?>').value = ""; // Clear the payment input box
                return;
            }

            //var amount = total - advance;
            var due = total - advance - (payment1 + payment2);
            // Set due to 0 if it's negative
            due = (due < 0 || due === total) ? 0 : due;
            // Update the value of the 'due' input box
            document.getElementById('due' + i).value = due.toFixed(2);
        }

        function checkallreadyComplete() {
            var temp_bill_id = document.getElementById('temp_bill_id').value;
            var table_id = document.getElementById('table_id').value;
            var room_id = document.getElementById('room_id').value;
            var type = 'check';
            if (temp_bill_id === '' && (table_id === '' || room_id === '')) {
                bootbox.alert('You Have Not Select Any Table Or Room');
            } else {
                $.get("checkAllreadycom.php?temp_bill_id=" + temp_bill_id + "&table_id=" + table_id + "&room_id=" + room_id + "&type=" + type).done(function(data) {
                    var obj = jQuery.parseJSON(data);
                    if (obj.status === 1) {
                        orderComplete(temp_bill_id, table_id, room_id);
                    } else {
                        bootbox.alert('Already Completed This Order');
                    }
                });
            }
        }

        function orderComplete(temp_bill_id, table_id, room_id) {
            var type = 'cusmealid';
            $.get("checkAllreadycom.php?temp_bill_id=" + temp_bill_id + "&table_id=" + table_id + "&room_id=" + room_id + "&type=" + type).done(function(data) {
                var obj = jQuery.parseJSON(data);
                if (obj.status === 0) {
                    if (table_id != '') {
                        location.href = "Invoice-18-" + table_id + ".html";
                    }
                    if (room_id !== '') {
                        location.href = "Invoicing_(Restaurant)-120-" + room_id + ".html";
                    }
                } else {
                    bootbox.alert('Can not Find  Your Invoice');
                }
            });
        }

        function validatePayment(id) {
            var payment = parseFloat(document.getElementById('payment' + id).value);
            var due = parseFloat(document.getElementById('due' + id).value);
            var error = document.getElementById('error' + id);

            if (document.getElementById('payment' + id).value.trim() !== "") {
                if (isNaN(payment) || payment <= 0) {
                    error.textContent = "Not a positive number";
                } else if (payment > due) {
                    error.textContent = "Payment cannot be greater than Due Amount";
                } else {
                    error.textContent = "";
                }
            } else {
                error.textContent = "";
            }
        }



        function submitDueForm(id) {
            var payment = document.getElementById('payment' + id).value;
            var error = document.getElementById('error' + id);
            if (payment === "") {
                error.textContent = "Payment Amount is required";
            } else {
                validatePayment(id);
                if (error.textContent === "") {

                    var staffID = document.getElementById('staffID' + id).value;
                    var due = document.getElementById('due' + id).value;
                    var payment = document.getElementById('payment' + id).value;

                    var req = getXmlHttpRequestObject();
                    if (req) {
                        req.onreadystatechange = function() {
                            if (req.readyState == 4) { //data is retrieved from server
                                if (req.status == 200) {
                                    var response = req.responseText;
                                    if (response.trim() === "true") {
                                        var newWindow = window.open("printStaffReciept.php?staffID=" + staffID + "&due=" + due + "&payment=" + payment, "_blank");

                                        var checkClosed = setInterval(function() {
                                            if (newWindow.closed) {
                                                clearInterval(checkClosed);
                                                location.reload();
                                            }
                                        }, 100);
                                    }
                                } else {
                                    alert("There was a problem while using XMLHTTP:\n");
                                }
                            }
                        }
                        var url = "updateDue.php?staffID=" + staffID + "&due=" + due + "&payment=" + payment;
                        req.open("GET", url, true);
                        req.send(null);
                    }
                }
            }
        }
